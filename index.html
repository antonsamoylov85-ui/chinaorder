// ChinaOrder — Google Apps Script // Привязан к таблице: открывать через Расширения → Apps Script const SHEET_RATES = 'Курсы'; const SHEET_BALANCES = 'Балансы'; const SHEET_ORDERS = 'Заявки'; // CORS — разрешаем запросы из браузера function doOptions(e) { return ContentService.createTextOutput('') .setMimeType(ContentService.MimeType.TEXT); } function doGet(e) { const action = e.parameter.action; const userId = e.parameter.userId; let result; switch(action) { case 'getData': result = getClientData(userId); break; case 'adminData': result = getAdminData(); break; case 'checkUser': result = checkUser(userId); break; default: result = { status: 'ok', message: 'ChinaOrder API работает!' }; } return jsonResponse(result); } function doPost(e) { let body; try { // Поддержка как application/json так и text/plain (no-cors из браузера) const raw = e.postData.contents; body = JSON.parse(raw); } catch(err) { return jsonResponse({ error: 'Invalid JSON' }); } let result; switch(body.action) { case 'syncData': result = syncData(body.orders, body.clients); break; case 'updateBalance': result = updateBalance(body.userId, body.amount); break; case 'newOrder': result = addOrder(body.order); break; case 'register': result = registerClient(body.profile); break; default: result = { error: 'Unknown action' }; } return jsonResponse(result); } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // Проверяет зарегистрирован ли пользователь по Telegram ID function checkUser(userId) { if (!userId) return { registered: false }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const balSheet = ss.getSheetByName(SHEET_BALANCES); if (!balSheet) return { registered: false }; const rows = balSheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(userId)) { const profile = { userId: String(rows[i][0]), lastName: rows[i][1] || '', firstName: rows[i][2] || '', middleName: rows[i][3] || '', phone: rows[i][4] || '', citizenship: rows[i][5] || '', citizenshipName: rows[i][5] || '', balRub: rows[i][6] || 0, registeredAt: rows[i][7] || '' }; return { registered: true, profile }; } } return { registered: false }; } function getClientData(userId) { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Курсы валют const ratesSheet = ss.getSheetByName(SHEET_RATES); let rates = {}; if (ratesSheet) { const data = ratesSheet.getDataRange().getValues(); // Пропускаем заголовок (строка 1 = ['Ключ','Значение','Обновлено']) for (let i = 1; i < data.length; i++) { if (data[i][0] && data[i][1]) { rates[data[i][0]] = parseFloat(data[i][1]); if (data[i][2]) rates.updated = String(data[i][2]); } } } // Баланс пользователя (колонка G = индекс 6) let balance = { rub: 0 }; if (userId && userId !== 'unknown') { const balSheet = ss.getSheetByName(SHEET_BALANCES); if (balSheet) { const rows = balSheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(userId)) { balance.rub = parseFloat(rows[i][6]) || 0; break; } } } } return { rates, balance }; } function getAdminData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const result = {}; const ratesSheet = ss.getSheetByName(SHEET_RATES); if (ratesSheet) { const data = ratesSheet.getDataRange().getValues(); const rates = {}; for (let i = 1; i < data.length; i++) { if (data[i][0] && data[i][1]) rates[data[i][0]] = parseFloat(data[i][1]); } result.rates = rates; } const balSheet = ss.getSheetByName(SHEET_BALANCES); if (balSheet) { const rows = balSheet.getDataRange().getValues(); result.clients = rows.slice(1).filter(r => r[0]).map(r => ({ userId: String(r[0]), lastName: r[1] || '', firstName: r[2] || '', middleName: r[3] || '', phone: r[4] || '', citizenshipName: r[5] || '', name: [r[1],r[2],r[3]].filter(Boolean).join(' ') || 'Клиент', balRub: parseFloat(r[6]) || 0, createdAt: r[7] ? new Date(r[7]).toISOString() : new Date().toISOString() })); } const ordSheet = ss.getSheetByName(SHEET_ORDERS); if (ordSheet) { const rows = ordSheet.getDataRange().getValues(); result.orders = rows.slice(1).filter(r => r[0]).map(r => ({ id: r[0], userId: String(r[1]), userName: r[2] || '', platform: r[3] || '', url: r[4] || '', description: r[5] || '', qty: r[6] || '1', budget: r[7] || '', comment: r[8] || '', status: r[9] || 'new', price: r[10] || null, createdAt: r[11] ? new Date(r[11]).toISOString() : new Date().toISOString() })); } return result; } function registerClient(profile) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let balSheet = ss.getSheetByName(SHEET_BALANCES); if (!balSheet) { balSheet = ss.insertSheet(SHEET_BALANCES); balSheet.getRange(1,1,1,8).setValues([[ 'Telegram ID','Фамилия','Имя','Отчество','Телефон','Гражданство','Баланс ₽','Дата регистрации' ]]); balSheet.getRange(1,1,1,8).setFontWeight('bold').setBackground('#fce8b2'); } const rows = balSheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(profile.userId)) { // Обновляем данные, баланс НЕ трогаем balSheet.getRange(i+1, 2, 1, 5).setValues([[ profile.lastName || '', profile.firstName || '', profile.middleName || '', profile.phone || '', profile.citizenshipName || '' ]]); return { success: true, updated: true }; } } // Новый клиент — баланс 0, пополняется вручную balSheet.appendRow([ profile.userId, profile.lastName || '', profile.firstName || '', profile.middleName || '', profile.phone || '', profile.citizenshipName || '', 0, new Date() ]); return { success: true, created: true }; } function updateBalance(userId, amount) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const balSheet = ss.getSheetByName(SHEET_BALANCES); if (!balSheet) return { error: 'Sheet not found' }; const rows = balSheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(userId)) { balSheet.getRange(i+1, 7).setValue(parseFloat(amount)); return { success: true, newBalance: amount }; } } return { error: 'User not found' }; } function syncData(orders, clients) { const ss = SpreadsheetApp.getActiveSpreadsheet(); if (clients && clients.length) { const balSheet = ss.getSheetByName(SHEET_BALANCES); if (balSheet) { const existing = balSheet.getDataRange().getValues(); clients.forEach(c => { for (let i = 1; i < existing.length; i++) { if (String(existing[i][0]) === String(c.userId)) { balSheet.getRange(i+1, 7).setValue(c.balRub); return; } } }); } } if (orders && orders.length) { const ordSheet = ss.getSheetByName(SHEET_ORDERS); if (ordSheet) { const existing = ordSheet.getDataRange().getValues(); orders.forEach(o => { for (let i = 1; i < existing.length; i++) { if (existing[i][0] === o.id) { ordSheet.getRange(i+1, 10).setValue(o.status); ordSheet.getRange(i+1, 11).setValue(o.price || ''); return; } } ordSheet.appendRow([ o.id, o.userId, o.userName, o.platform, o.url, o.description, o.qty, o.budget, o.comment, o.status, o.price || '', new Date(o.createdAt) ]); }); } } return { success: true }; } function addOrder(order) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let ordSheet = ss.getSheetByName(SHEET_ORDERS); if (!ordSheet) { ordSheet = ss.insertSheet(SHEET_ORDERS); ordSheet.getRange(1,1,1,12).setValues([[ 'ID Заявки','User ID','Имя','Площадка','Ссылка', 'Описание','Кол-во','Бюджет ¥','Комментарий', 'Статус','Цена ¥','Дата' ]]); ordSheet.getRange(1,1,1,12).setFontWeight('bold').setBackground('#fce8b2'); } ordSheet.appendRow([ order.id, order.userId, order.userName, order.platform, order.url, order.description, order.qty, order.budget, order.comment, 'new', '', new Date() ]); return { success: true }; } // ══════════════════════════════════════════════════ // ЗАПУСТИТЕ ОДИН РАЗ для создания листов // ══════════════════════════════════════════════════ function initSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet();
